\directlua{
function Escape(s)
    return "'" .. s:gsub("'", [['"'"']]) .. "'"
end
function Popen(cmd)
    local handle, err = io.popen(cmd)
    if not handle then
        print(err)
        os.exit(1)
    end
    return handle
end
function PrintFile(file)
    for line in file:lines() do
        tex.print(line)
    end
end
}

\newcommand{\LirstInline}[2]{\ignorespaces\directlua{
    local handle = Popen('lirstings inline ' .. Escape([[\unexpanded{#1}]]) .. ' ' .. Escape([[\unexpanded{#2}]]))
    tex.print(handle:read('*all'))
    handle:close()
}\unskip}

\newcommand{\Lirsting}[2][]{\directlua{
    local handle = Popen('lirstings from-tex ' .. Escape([[\unexpanded{#2}]]) .. ' ' .. Escape([[\unexpanded{#1}]]))
    PrintFile(handle)
    handle:close()
}}
